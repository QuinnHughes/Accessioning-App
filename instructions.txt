def get_input(prompt, default=None):
    """Get user input with optional default value."""
    if default is not None:
        user_input = input(f"{prompt} [default: {default}]: ").strip()
        return user_input if user_input else default
    return input(f"{prompt}: ").strip()

def get_int_input(prompt, default=None):
    """Get integer input with validation."""
    while True:
        try:
            value = get_input(prompt, default)
            return int(value)
        except ValueError:
            print("Please enter a valid number.")

def generate_labels():
    """Generate inventory labels based on user input."""
    print("\n=== INVENTORY LABEL GENERATOR ===\n")
    
    # Get prefix
    prefix = get_input("Enter prefix (e.g., S-1-01B)", "S-1-01B")
    
    # Get ladder range
    start_ladder = get_int_input("Start ladder", 1)
    end_ladder = get_int_input("End ladder", start_ladder)
    
    # Get shelf range
    start_shelf = get_int_input("Start shelf", 1)
    end_shelf = get_int_input("End shelf", start_shelf)
    
    # Ask for specific items or full range
    print("\nItem selection:")
    print("  1. Specific items (e.g., 3, 15, 19)")
    print("  2. Full range (e.g., 1-100)")
    choice = get_input("Choose option (1 or 2)", "1")
    
    specific_items_list = []
    items_per_shelf = 100
    
    if choice == "1":
        specific_items = get_input("Enter item numbers (comma-separated, e.g., 3, 15, 19)", "")
        if specific_items:
            specific_items_list = [int(item.strip()) for item in specific_items.split(",")]
    else:
        items_per_shelf = get_int_input("Items per shelf", 100)
    
    # Generate call numbers
    call_numbers = []
    for ladder in range(start_ladder, end_ladder + 1):
        for shelf in range(start_shelf, end_shelf + 1):
            if specific_items_list:
                for item in specific_items_list:
                    call_number = f"{prefix}-{ladder:02}-{shelf:02}-{item:03}"
                    call_numbers.append(call_number)
            else:
                for item in range(1, items_per_shelf + 1):
                    call_number = f"{prefix}-{ladder:02}-{shelf:02}-{item:03}"
                    call_numbers.append(call_number)
    
    # Format output
    output = ""
    for call_number in call_numbers:
        output += f"{call_number}\n\n\n===============\n"
    
    # Display results
    print(f"\n✓ Generated {len(call_numbers)} labels")
    print("\n" + output)
    
    return output

# Main loop
while True:
    generate_labels()
    
    # Ask if user wants to generate more
    again = get_input("\nGenerate more labels? (y/n)", "y").lower()
    if again != 'y':
        print("\nGoodbye!")
        break
